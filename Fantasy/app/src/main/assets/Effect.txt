class ActionInfo{
	//操作中か		isProcess:boolean
	//攻撃者		owner:Actor
	//被攻撃者		target:Actor
	//攻撃タイプ	action_type:ActionType
	//ID			id:int
	//ダメージ		damage:int
}

//Enemyのaction
void action(ActionInfo info){
	//敵の場合は常にfalse
	info.isProcess = false;
	//攻撃者は自分
	info.owner = this;
	//被攻撃者の決定
	info.target = choiceTarget();
	//infoの攻撃タイプとIDを決定
	choiceAction(info);	
	//ダメージの計算	自分を渡すと相手側で計算してくれる
	info.damage = info.target.calcDamage(this);
}

//エフェクトの表示
class EffectDrawMachine(){
	protected boolean isEnd = false;

	//エフェクトを表示するクラス
	void drawEffect(ActionInfo info){
		float x = 0,y = 0,
		float sx = 1,sy = 1;
		if(info.target instanceof Player){
			//プレイヤーに対しての行動
			//エフェクトの表示位置を画面中央に
			//さらにエフェクトの大きさを大きくする
			x = GLES20Util.getWidth_gl() / 2f;
			y = GLES20Util.getHeight_gl() / 2f;
			sx = 2;
			sy = 2;
		}else if(info.target instanceof Enemy){
			//敵に対しての行動
			x = info.target.getX();
			y = info.target.getY();
			sx = 1;
			sy = 1;
		}
		//エフェクトの表示
		isEnd = info.owner.drawSkillEffect(info);
		
	}
	public boolean isEnd(){
		return isEnd;
	}

}
